SDKs & Libraries

Integrate CoinFuse into your application using our official SDKs and community-maintained libraries.

Official SDKs

Python

pip install coinfuse-python 
from coinfuse import CoinFuse

client = CoinFuse('your-api-key')

# Create a payment
payment = client.payments.create(
    amount=100,
    currency='USD',
    crypto_currency='BTC',
    callback_url='https://your-site.com/webhook'
)

# Handle webhooks
@app.route('/webhook', methods=['POST'])
def webhook_handler():
    event = client.webhooks.construct_event(
        request.data,
        request.headers['CoinFuse-Signature'],
        'your-webhook-secret'
    )
    # Handle the event
    return '', 200

JavaScript

npm install @coinfuse/js
yarn add @coinfuse/js 
import { CoinFuse } from '@coinfuse/js';

const client = new CoinFuse('your-api-key');

// Create a payment
const payment = await client.payments.create({
    amount: 100,
    currency: 'USD',
    cryptoCurrency: 'BTC',
    callbackUrl: 'https://your-site.com/webhook'
});

// Handle webhooks
app.post('/webhook', (req, res) => {
    const event = client.webhooks.constructEvent(
        req.body,
        req.headers['coinfuse-signature'],
        'your-webhook-secret'
    );
    // Handle the event
    res.sendStatus(200);
});

Ruby

gem install coinfuse 
require 'coinfuse'

client = CoinFuse::Client.new('your-api-key')

# Create a payment
payment = client.payments.create(
    amount: 100,
    currency: 'USD',
    crypto_currency: 'BTC',
    callback_url: 'https://your-site.com/webhook'
)

# Handle webhooks
post '/webhook' do
    event = client.webhooks.construct_event(
        request.body.read,
        request.env['HTTP_COINFUSE_SIGNATURE'],
        'your-webhook-secret'
    )
    # Handle the event
    status 200
end
