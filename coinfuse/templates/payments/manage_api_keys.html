from cryptostripe import CryptoStripe

client = CryptoStripe('your-api-key')

# Create a payment
payment = client.Payment.create(
    amount_usd=10.00,
    crypto_type='BTC'
)

# Get payment status
status = client.Payment.get(payment.id)

# Create a subscription
subscription = client.Subscription.create(
    plan_name='Pro Plan',
    amount_usd=49.99,
    crypto_type='ETH',
    interval='monthly'
)
